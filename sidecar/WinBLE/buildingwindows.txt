$project = "C:\Users\TJ\Documents\Code\go\meshexec"
dotnet build $project/sidecar/WinBLE/WinBLE.csproj -c Release                     
dotnet publish $project/sidecar/WinBLE/WinBLE.csproj -c Release -r win-x64 --self-contained true -p:PublishSingleFile=true -p:IncludeAllContentForSelfExtract=true

Copy-Item "$project/sidecar/WinBLE/bin/Release/net8.0-windows10.0.19041.0/win-x64/publish/*" "$project/sidecar/WinBLE/msix/WinBLEPackage" -Recurse -Force
$pkgDir = "$project/sidecar/WinBLE/msix/WinBLEPackage"
$msix   = "$project/sidecar/WinBLE/msix/WinBLE.msix"


$makeappx = Get-ChildItem "C:\Program Files (x86)\Windows Kits\10\bin" -Recurse -Filter makeappx.exe -ErrorAction SilentlyContinue | Sort-Object FullName -Descending | Select-Object -First 1 -ExpandProperty FullName
& "$makeappx" pack /d "$pkgDir" /p "$msix" /l


$signtool = Get-ChildItem "C:\Program Files (x86)\Windows Kits\10\bin" -Recurse -Filter signtool.exe -ErrorAction SilentlyContinue | Sort-Object FullName -Descending | Select-Object -First 1 -ExpandProperty FullName
& "$signtool" sign /fd SHA256 /f "$env:TEMP\MeshExecDev.pfx" /p "meshExecDevP@ss" "$msix"
& "$signtool" verify /pa /v "$msix"

Get-AppxPackage Meshexec.WinBLE | Remove-AppxPackage
Add-AppxPackage -Path "$msix" -ForceUpdateFromAnyVersion -ForceApplicationShutdown

$p = Get-AppxPackage -Name Meshexec.WinBLE -AllUsers
$p | Format-List Name, PackageFamilyName, PackageFullName

explorer.exe "shell:AppsFolder\$($p.PackageFamilyName)!App"

$log = Join-Path (Join-Path $env:LOCALAPPDATA "Packages\$($p.PackageFamilyName)\LocalState") "WinBLE.log"
$log
Get-Content -Path $log -Wait


Join-Path $p.InstallLocation 'WinBLE.exe'
